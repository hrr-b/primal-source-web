SECTION 1 — USER PROFILE MODEL
-----------------------------------

Create: /lib/userProfile.ts

export function getUserProfile() {
  const profile = JSON.parse(localStorage.getItem("user_profile") || "{}");

  return {
    visits: profile.visits || 0,
    clicks: profile.clicks || 0,
    scrollDepth: profile.scrollDepth || 0,
    timeOnSite: profile.timeOnSite || 0,
    intentScore: profile.intentScore || 0
  };
}

export function updateUserProfile(update) {
  const current = getUserProfile();
  const updated = { ...current, ...update };

  localStorage.setItem("user_profile", JSON.stringify(updated));
}

SECTION 2 — INTENT SCORING SYSTEM
-----------------------------------

Create dynamic score:

export function calculateIntent(profile) {
  let score = 0;

  if (profile.visits > 1) score += 2;
  if (profile.clicks > 3) score += 3;
  if (profile.scrollDepth > 70) score += 2;
  if (profile.timeOnSite > 30) score += 3;

  return score;
}

SECTION 4 — ADAPTIVE LOADER
-----------------------------------

Create: /lib/adaptiveContent.ts

import variants from "../content/homepage_variants.json";

export function getAdaptiveContent(score) {
  if (score >= 7) return variants.high_intent;
  if (score >= 4) return variants.mid_intent;
  return variants.low_intent;
}

SECTION 6 — NEXT.JS IMPLEMENTATION
-----------------------------------

"use client";

import { useEffect, useState } from "react";
import { getUserProfile, updateUserProfile } from "@/lib/userProfile";
import { calculateIntent } from "@/lib/intent";
import { getAdaptiveContent } from "@/lib/adaptiveContent";

export default function Home() {
  const [content, setContent] = useState(null);

  useEffect(() => {
    const profile = getUserProfile();
    const score = calculateIntent(profile);
    const adaptive = getAdaptiveContent(score);

    setContent(adaptive);

    updateUserProfile({
      visits: profile.visits + 1
    });
  }, []);

  if (!content) return null;

  return <h1>{content.hero.headline}</h1>;
}
